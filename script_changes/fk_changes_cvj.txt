CREATE TABLE cvjdb.inventory_category
(
    category_id INT(11) PRIMARY KEY AUTO_INCREMENT,
    category_name TEXT
);

ALTER TABLE cvjdb.inventory
ADD CONSTRAINT inventory_inventory_category_category_id_fk
FOREIGN KEY (category) REFERENCES inventory_category (category_id);

CREATE TABLE cvjdb.package_inventory
(
    package_id INT(11),
    inventory_id INT(5),
    category_id INT(11),
    rent_cost DECIMAL(10),
    quantity INT(11),
    CONSTRAINT package_inventory_package_package_id_fk FOREIGN KEY (package_id) REFERENCES package (package_id),
    CONSTRAINT package_inventory_inventory_inventory_id_fk FOREIGN KEY (inventory_id) REFERENCES inventory (inventory_id)
);

ALTER TABLE cvjdb.inventory ADD rental_cost DECIMAL(10) DEFAULT 0 NULL;

CREATE TABLE cvjdb.package_misc_item
(
    id INT PRIMARY KEY AUTO_INCREMENT,
    package_id INT(11),
    name TEXT,
    quantity INT,
    unit_cost DECIMAL(10),
    CONSTRAINT package_misc_item_package_package_id_fk FOREIGN KEY (package_id) REFERENCES package (package_id)
);

ALTER TABLE cvjdb.event_budget_item ADD item_tag TEXT NULL;
